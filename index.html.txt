    <script>
        AOS.init({ duration: 1000, once: true });

        const contactForm = document.getElementById('contactForm');
        const toast = document.getElementById('toast');
        const submitBtn = document.getElementById('submitBtn');

        contactForm.addEventListener('submit', async function(e) {
            e.preventDefault();
            
            // تم وضع الرابط الخاص بك هنا مباشرة
            const url = "https://formspree.io/f/xrelvdgd";

            submitBtn.disabled = true;
            submitBtn.innerText = "جاري الإرسال...";

            const formData = new FormData(this);
            const data = Object.fromEntries(formData.entries());

            try {
                const response = await fetch(url, {
                    method: 'POST',
                    body: JSON.stringify(data),
                    headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' }
                });

                if (response.ok) {
                    showToast("تم الإرسال بنجاح! تفقد بريدك الإلكتروني.");
                    this.reset();
                } else {
                    showToast("حدث خطأ، يرجى المحاولة لاحقاً.");
                }
            } catch (error) {
                showToast("فشل الاتصال، تحقق من الإنترنت.");
            } finally {
                submitBtn.disabled = false;
                submitBtn.innerText = "إرسال المعلومات";
            }
        });

        function showToast(message) {
            toast.innerText = message;
            toast.classList.add('active');
            setTimeout(() => toast.classList.remove('active'), 4000);
        }
    </script>
